void uartinit(void)
{
	UBRR0L=0x0C;
	UCSR0B |= (1 << RXEN0 ) | (1 << TXEN0 )| (1<<RXCIE0);

}


uart_trans(char tx_data)
{
	while (( UCSR0A & (1 << UDRE0 )) == 0);
	UDR0 = tx_data ;
	
}

void stransmit0(char *p)
{
	while(*p) uart_trans(*p++);
}

lcdinit()

}

uint8_t get_adc(unsigned char ch)
{
	ADMUX=0x60|ch;
	ADCSRA=0xC5;
	while(!ADIF);
	_delay_ms(10);
	return(ADCH);
}

void cmd(unsigned char c)
{
	PORTC=c;
	PORTD=0x20;
	_delay_ms(1);
	PORTD=0x00;
}

void data(unsigned	char c)
{
	PORTC=c;
	_delay_ms(1);
	PORTD=0xA0;
	_delay_ms(1);
	PORTD=0x80;
	_delay_ms(1);
	cmd(0x06);
}

void string(char *p)
{
	while(*p) data(*p++);
	
}
